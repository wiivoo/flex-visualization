<objective>
Build the pure-logic battery optimizer: `runBatteryDay` (greedy three-pass, 96-slot day) and `runBatteryYear` (annual roll-up). The optimizer is the correctness-critical core of the phase — if it gets the DE grid-export prohibition wrong, every downstream component displays misleading numbers.

This plan uses test-first development because the business rules are crisp:
- DE export prohibition: `gridExportKwh === 0` for every slot, in every day, in every scenario.
- SoC bounds: `0 ≤ socKwh ≤ usableKwh` at all times.
- Energy conservation: per-slot load is served by (PV direct) + (battery discharge) + (grid import) with no discrepancy.
- Round-trip efficiency applied on the charge→discharge path.
- Standby power consumption always deducted.

The optimizer runs client-side (<5ms for 96 slots). Pure TypeScript, no async, no I/O, no external libraries.

import type { HourlyPrice } from '@/lib/v2-config'

export interface BatteryParams {
  usableKwh: number
  maxChargeKw: number          // kW (AC-side) — applied per-slot as kW × slotHours
  maxDischargeKw: number
  roundTripEff: number         // 0..1, AC-to-AC
  standbyWatts: number         // continuous parasitic draw
  feedInCapKw: number          // hard cap on feed-in (DE 0.8, user-toggle 2.0)
  allowGridExport: boolean     // false for Phase 8 in both DE and NL (plug-in regime)
  buyPriceCtKwh?: (daPrice: number) => number   // optional markup function; default: identity
  sellPriceCtKwh?: (daPrice: number) => number  // optional; returns 0 when allowGridExport=false
}

export interface SlotResult {
  timestamp: number
  hour: number
  minute: number
  priceCtKwh: number           // day-ahead price (ct/kWh)
  pvKwh: number                // PV generation this slot
  loadKwh: number              // household load this slot
  pvSelfKwh: number            // PV used directly by load (not via battery)
  chargeFromPvKwh: number
  chargeFromGridKwh: number
  dischargeToLoadKwh: number
  gridImportKwh: number        // always >= 0 (never negative — no export under plug-in regime)
  gridExportKwh: number        // always 0 in Phase 8 (DE prohibited; NL plug-in regime)
  socKwhStart: number
  socKwhEnd: number
  slotCostEur: number          // grid cost for this slot in EUR (baseline comparison lives in summary)
  baselineCostEur: number      // what the grid cost would be with no battery (load × priceCtKwh / 100)
}

export interface DaySummary {
  baselineCostEur: number      // sum of slotBaselineCost (no-battery scenario)
  optimizedCostEur: number     // sum of slotCost + standbyCost (with-battery scenario)
  savingsEur: number           // baseline - optimized
  arbitrageSavingsEur: number  // portion of savings from charge-cheap / discharge-expensive
  pvSelfConsumptionValueEur: number  // portion of savings from avoiding grid import via direct PV + PV-to-battery-to-load
  standbyCostEur: number       // total standby for the day (always >= 0)
  gridImportKwh: number
  gridExportKwh: number        // always 0
  batteryCyclesEquivalent: number  // (sum of charge kWh) / usableKwh
}

export interface MonthlyBatteryResult {
  month: string                // 'YYYY-MM'
  savingsEur: number
  arbitrageSavingsEur: number
  pvSelfConsumptionValueEur: number
  standbyCostEur: number
  cyclesEquivalent: number
}

export interface AnnualBatteryResult {
  annualSavingsEur: number
  arbitrageSavingsEur: number
  pvSelfConsumptionValueEur: number
  standbyCostEur: number
  gridImportKwh: number
  cyclesEquivalent: number
  months: MonthlyBatteryResult[]
}

export function runBatteryDay(
  prices: HourlyPrice[],
  pvKwhPerSlot: number[],      // same length as prices
  loadKwhPerSlot: number[],    // same length as prices
  params: BatteryParams,
  startSocKwh?: number,        // default 0
): { slots: SlotResult[]; summary: DaySummary }

export function runBatteryYear(
  pricesByDate: Map<string, HourlyPrice[]>,   // key: 'YYYY-MM-DD'
  pvProfile: number[],                         // 8760 values, normalized sum = 1.0
  loadProfile: number[],                       // 8760 values, normalized sum = 1.0
  pvKwhPerYear: number,                        // scale factor for PV (0 if no PV)
  annualLoadKwh: number,
  params: BatteryParams,
): AnnualBatteryResult

import { describe, it, expect } from 'vitest'
import { runBatteryDay, runBatteryYear, type BatteryParams } from '@/lib/battery-optimizer'
import type { HourlyPrice } from '@/lib/v2-config'

function mkHourlyPrices(pricesCt: number[]): HourlyPrice[] {
  // Build 96 QH slots from a pattern of 24 hourly prices (repeat each 4×)
  // Timestamp: 2025-06-15 UTC
  const slots: HourlyPrice[] = []
  const baseTs = Date.UTC(2025, 5, 15, 0, 0, 0)
  for (let i = 0; i < 96; i++) {
    const hour = Math.floor(i / 4)
    slots.push({
      timestamp: baseTs + i * 15 * 60_000,
      startMs: baseTs + i * 15 * 60_000,
      endMs: baseTs + (i + 1) * 15 * 60_000,
      price: pricesCt[hour] * 10,        // EUR/MWh (matches HourlyPrice convention from v2-config)
      ctPerKwh: pricesCt[hour],
    } as HourlyPrice)
  }
  return slots
}

function mkUniformLoad(kwhPerDay: number): number[] {
  return Array(96).fill(kwhPerDay / 96)
}

function mkNoPv(): number[] { return Array(96).fill(0) }

const BASE_PARAMS: BatteryParams = {
  usableKwh: 2.0,
  maxChargeKw: 1.5,
  maxDischargeKw: 0.8,
  roundTripEff: 0.88,
  standbyWatts: 10,
  feedInCapKw: 0.8,
  allowGridExport: false,
}

describe('runBatteryDay — DE export prohibition', () => {
  it('every slot has gridExportKwh === 0 regardless of price spread', () => {
    const prices = mkHourlyPrices([
      // Cheap midnight, expensive evening — big arbitrage incentive
      2,2,2,2, 3,3,5,8, 12,15,18,20, 22,25,28,30, 35,40,45,50, 45,30,15,8,
    ])
    const { slots } = runBatteryDay(prices, mkNoPv(), mkUniformLoad(5), BASE_PARAMS)
    for (const s of slots) expect(s.gridExportKwh).toBe(0)
  })

  it('gridImportKwh is always >= 0', () => {
    const prices = mkHourlyPrices(Array(24).fill(0).map((_,h) => h * 2))  // monotonic
    const { slots } = runBatteryDay(prices, mkNoPv(), mkUniformLoad(5), BASE_PARAMS)
    for (const s of slots) expect(s.gridImportKwh).toBeGreaterThanOrEqual(-0.001)
  })

  it('battery never discharges more than loadKwh in any slot (self-consumption only)', () => {
    const prices = mkHourlyPrices([
      50,50,50,50, 50,50,50,50, 50,50,50,50, 50,50,50,50, 50,50,50,50, 50,50,50,50
    ])  // flat high price — battery would want to export if allowed
    const { slots } = runBatteryDay(prices, mkNoPv(), mkUniformLoad(3), BASE_PARAMS)
    for (const s of slots) {
      expect(s.dischargeToLoadKwh).toBeLessThanOrEqual(s.loadKwh + 0.001)
    }
  })
})

describe('runBatteryDay — SoC bounds', () => {
  it('socKwhEnd is always in [0, usableKwh]', () => {
    const prices = mkHourlyPrices([
      2,2,2,2, 3,3,5,8, 12,15,18,20, 22,25,28,30, 35,40,45,50, 45,30,15,8,
    ])
    const { slots } = runBatteryDay(prices, mkNoPv(), mkUniformLoad(5), BASE_PARAMS, 0)
    for (const s of slots) {
      expect(s.socKwhEnd).toBeGreaterThanOrEqual(-0.01)
      expect(s.socKwhEnd).toBeLessThanOrEqual(BASE_PARAMS.usableKwh + 0.01)
    }
  })

  it('socKwhStart of slot n+1 equals socKwhEnd of slot n', () => {
    const prices = mkHourlyPrices([
      2,2,2,2, 3,3,5,8, 12,15,18,20, 22,25,28,30, 35,40,45,50, 45,30,15,8,
    ])
    const { slots } = runBatteryDay(prices, mkNoPv(), mkUniformLoad(5), BASE_PARAMS, 0.5)
    for (let i = 1; i < slots.length; i++) {
      expect(slots[i].socKwhStart).toBeCloseTo(slots[i-1].socKwhEnd, 3)
    }
    expect(slots[0].socKwhStart).toBe(0.5)
  })
})

describe('runBatteryDay — energy conservation', () => {
  it('per-slot: loadKwh === pvSelfKwh + dischargeToLoadKwh + gridImportKwh', () => {
    const prices = mkHourlyPrices([
      2,2,2,2, 3,3,5,8, 12,15,18,20, 22,25,28,30, 35,40,45,50, 45,30,15,8,
    ])
    // Moderate PV, midday bell
    const pv = Array(96).fill(0).map((_, i) => {
      const h = i / 4
      return Math.max(0, Math.sin((h - 6) / 14 * Math.PI)) * 0.05
    })
    const { slots } = runBatteryDay(prices, pv, mkUniformLoad(5), { ...BASE_PARAMS, maxDischargeKw: 1.0 })
    for (const s of slots) {
      const delivered = s.pvSelfKwh + s.dischargeToLoadKwh + s.gridImportKwh
      expect(delivered).toBeCloseTo(s.loadKwh, 3)
    }
  })
})

describe('runBatteryDay — round-trip efficiency', () => {
  it('total battery-out ≈ total battery-in × roundTripEff (daily rollup)', () => {
    const prices = mkHourlyPrices([
      2,2,2,2, 2,2,2,2, 2,2,2,2, 40,40,40,40, 40,40,40,40, 40,40,40,40,
    ])  // half-day cheap, half-day expensive
    const { slots } = runBatteryDay(prices, mkNoPv(), mkUniformLoad(8), BASE_PARAMS)
    const chargeIn = slots.reduce((a,s) => a + s.chargeFromGridKwh + s.chargeFromPvKwh, 0)
    const dischargeOut = slots.reduce((a,s) => a + s.dischargeToLoadKwh, 0)
    if (chargeIn > 0.1) {
      expect(dischargeOut).toBeLessThanOrEqual(chargeIn * BASE_PARAMS.roundTripEff + 0.05)
    }
  })
})

describe('runBatteryDay — guards', () => {
  it('returns zeroed result when usableKwh === 0', () => {
    const prices = mkHourlyPrices(Array(24).fill(20))
    const r = runBatteryDay(prices, mkNoPv(), mkUniformLoad(5), { ...BASE_PARAMS, usableKwh: 0 })
    expect(r.summary.savingsEur).toBe(0)
    for (const s of r.slots) {
      expect(s.chargeFromGridKwh).toBe(0)
      expect(s.dischargeToLoadKwh).toBe(0)
      expect(s.gridExportKwh).toBe(0)
    }
  })

  it('returns zeroed result when maxChargeKw === 0', () => {
    const prices = mkHourlyPrices(Array(24).fill(20))
    const r = runBatteryDay(prices, mkNoPv(), mkUniformLoad(5), { ...BASE_PARAMS, maxChargeKw: 0 })
    expect(r.summary.savingsEur).toBe(0)
  })

  it('does not produce negative savings vs baseline', () => {
    // Flat price day — no arbitrage opportunity
    const prices = mkHourlyPrices(Array(24).fill(20))
    const r = runBatteryDay(prices, mkNoPv(), mkUniformLoad(5), BASE_PARAMS)
    // Optimizer may idle; standby cost can push into negative, but arbitrageSavings must be >= 0
    expect(r.summary.arbitrageSavingsEur).toBeGreaterThanOrEqual(-0.001)
  })
})

describe('runBatteryDay — standby cost', () => {
  it('accrues 10W × 24h × avgPrice for the day with no cycling', () => {
    // Flat price day, no PV, 10W standby, no arbitrage — standbyCost should be 10W × 24h × 20ct/100 / 1000 = 0.048 EUR
    const prices = mkHourlyPrices(Array(24).fill(20))
    const r = runBatteryDay(prices, mkNoPv(), mkUniformLoad(5), BASE_PARAMS)
    expect(r.summary.standbyCostEur).toBeCloseTo(10 * 24 / 1000 * 20 / 100, 2)
  })
})

describe('runBatteryYear — annual roll-up', () => {
  it('aggregates 3 days × flat prices into a non-negative annual savings', () => {
    const dates = ['2025-06-15', '2025-06-16', '2025-06-17']
    const map = new Map<string, HourlyPrice[]>()
    for (const d of dates) {
      const [y,m,day] = d.split('-').map(Number)
      const ts = Date.UTC(y, m-1, day)
      const slots: HourlyPrice[] = []
      for (let i = 0; i < 96; i++) {
        const hour = Math.floor(i / 4)
        slots.push({
          timestamp: ts + i * 15 * 60_000,
          startMs: ts + i * 15 * 60_000,
          endMs: ts + (i + 1) * 15 * 60_000,
          price: (hour < 6 ? 5 : hour > 18 ? 40 : 20) * 10,
          ctPerKwh: hour < 6 ? 5 : hour > 18 ? 40 : 20,
        } as HourlyPrice)
      }
      map.set(d, slots)
    }
    const pvProfile = Array(8760).fill(1/8760)
    const loadProfile = Array(8760).fill(1/8760)
    const r = runBatteryYear(map, pvProfile, loadProfile, 0, 2500, BASE_PARAMS)
    expect(r.annualSavingsEur).toBeGreaterThanOrEqual(0)
    expect(r.months.length).toBeGreaterThan(0)
    expect(r.months.every(m => /^\d{4}-\d{2}$/.test(m.month))).toBe(true)
  })
})

/**
 * Battery day + annual optimizer — greedy three-pass SoC-constrained schedule.
 *
 * Hard constraints (validated by test suite):
 *  - Grid export is ALWAYS 0 (Phase 8 plug-in regime — DE prohibited by VDE-AR-N 4105:2026-03,
 *    NL plug-in regime is self-consumption-only at the modelled inverter sizes).
 *  - SoC always in [0, usableKwh].
 *  - Per-slot: loadKwh = pvSelfKwh + dischargeToLoadKwh + gridImportKwh (conservation).
 *  - Round-trip efficiency applied on charge→discharge path.
 *  - Standby parasitic draw accrues every slot.

function zeroedDay(prices: HourlyPrice[]): { slots: SlotResult[]; summary: DaySummary } {
  const slots = prices.map(p => ({
    timestamp: p.timestamp, hour: new Date(p.timestamp).getUTCHours(),
    minute: new Date(p.timestamp).getUTCMinutes(),
    priceCtKwh: Number((p as any).ctPerKwh ?? p.price / 10),
    pvKwh: 0, loadKwh: 0, pvSelfKwh: 0,
    chargeFromPvKwh: 0, chargeFromGridKwh: 0, dischargeToLoadKwh: 0,
    gridImportKwh: 0, gridExportKwh: 0,
    socKwhStart: 0, socKwhEnd: 0,
    slotCostEur: 0, baselineCostEur: 0,
  }))
  return {
    slots,
    summary: {
      baselineCostEur: 0, optimizedCostEur: 0, savingsEur: 0,
      arbitrageSavingsEur: 0, pvSelfConsumptionValueEur: 0, standbyCostEur: 0,
      gridImportKwh: 0, gridExportKwh: 0, batteryCyclesEquivalent: 0,
    }
  }
}

export function runBatteryDay(
  prices: HourlyPrice[],
  pvKwhPerSlot: number[],
  loadKwhPerSlot: number[],
  params: BatteryParams,
  startSocKwh: number = 0,
): { slots: SlotResult[]; summary: DaySummary } {
  // Guard
  if (params.usableKwh <= 0 || params.maxChargeKw <= 0) return zeroedDay(prices)
  if (prices.length === 0) return zeroedDay(prices)
  if (pvKwhPerSlot.length !== prices.length || loadKwhPerSlot.length !== prices.length) {
    throw new Error(`runBatteryDay: length mismatch — prices=${prices.length}, pv=${pvKwhPerSlot.length}, load=${loadKwhPerSlot.length}`)
  }

  const N = prices.length
  const slotHours = (prices[0].endMs && prices[0].startMs) ? (prices[0].endMs - prices[0].startMs) / 3_600_000 : (24 / N)
  const maxChargePerSlot = params.maxChargeKw * slotHours
  const maxDischargePerSlot = params.maxDischargeKw * slotHours

  // ---- Pass 1: build per-slot baseline + PV direct self-consumption ----
  // PV direct to load: min(pv, load) — this is always the best choice (free energy).
  // PV surplus: candidate for battery charging in pass 2.
  const slots: SlotResult[] = new Array(N)
  for (let i = 0; i < N; i++) {
    const priceCtKwh = Number((prices[i] as any).ctPerKwh ?? prices[i].price / 10)
    const pvKwh = Math.max(0, pvKwhPerSlot[i])
    const loadKwh = Math.max(0, loadKwhPerSlot[i])
    const pvSelfKwh = Math.min(pvKwh, loadKwh)
    const baselineGridKwh = Math.max(0, loadKwh - pvSelfKwh)  // grid needed without battery
    slots[i] = {
      timestamp: prices[i].timestamp,
      hour: new Date(prices[i].timestamp).getUTCHours(),
      minute: new Date(prices[i].timestamp).getUTCMinutes(),
      priceCtKwh,
      pvKwh, loadKwh, pvSelfKwh,
      chargeFromPvKwh: 0,
      chargeFromGridKwh: 0,
      dischargeToLoadKwh: 0,
      gridImportKwh: baselineGridKwh,
      gridExportKwh: 0,
      socKwhStart: 0,
      socKwhEnd: 0,
      slotCostEur: baselineGridKwh * priceCtKwh / 100,
      baselineCostEur: baselineGridKwh * priceCtKwh / 100,
    }
  }

  // ---- Pass 2: PV surplus → battery charge (free arbitrage) ----
  //                then grid arbitrage (charge cheap, discharge expensive to load)
  //
  // Chronological walk. Maintain a running SoC. For each slot:
  //   1. If there's PV surplus (pv > load), try to charge battery from surplus up to maxChargePerSlot AND SoC headroom.
  //   2. After charging from PV, if price is in "cheap" quantile, top up from grid up to budget.
  //   3. If price is in "expensive" quantile and SoC > 0, discharge to load up to maxDischargePerSlot AND loadKwh ceiling.
  //
  // Cheap / expensive thresholds: sort slot prices; take 40th percentile = cheap cutoff, 60th percentile = expensive cutoff.
  // This heuristic approximates the optimal greedy pairing without full DP.

  const sortedPrices = slots.map(s => s.priceCtKwh).slice().sort((a, b) => a - b)
  const cheapCutoff = sortedPrices[Math.floor(N * 0.40)]
  const expensiveCutoff = sortedPrices[Math.floor(N * 0.60)]

  let socKwh = Math.max(0, Math.min(startSocKwh, params.usableKwh))

  for (let i = 0; i < N; i++) {
    const s = slots[i]
    s.socKwhStart = socKwh

    // (1) PV surplus → battery
    const pvSurplus = Math.max(0, s.pvKwh - s.pvSelfKwh)
    const headroom = params.usableKwh - socKwh
    const canChargeFromPv = Math.min(pvSurplus, maxChargePerSlot, headroom)
    if (canChargeFromPv > 0) {
      s.chargeFromPvKwh = canChargeFromPv
      socKwh += canChargeFromPv
    }

    // (2) Grid charge when cheap
    const chargeBudget = Math.min(
      maxChargePerSlot - s.chargeFromPvKwh,
      params.usableKwh - socKwh,
    )
    if (chargeBudget > 0 && s.priceCtKwh <= cheapCutoff) {
      s.chargeFromGridKwh = chargeBudget
      socKwh += chargeBudget
      s.gridImportKwh += chargeBudget   // additional grid draw for charging
      s.slotCostEur = s.gridImportKwh * s.priceCtKwh / 100
    }

    // (3) Discharge to load when expensive
    if (s.priceCtKwh >= expensiveCutoff && socKwh > 0) {
      // Available energy out = socKwh * roundTripEff (apply half of RTE penalty on discharge)
      // Convention: apply full RTE on discharge side (energy out = battery draw × RTE).
      // Limit by load this slot (no export under plug-in regime) and max discharge power.
      const maxOut = Math.min(
        maxDischargePerSlot,
        s.loadKwh - s.pvSelfKwh,        // only replace residual load (conservation)
        socKwh * params.roundTripEff,
        params.feedInCapKw * slotHours, // hardware feed-in cap
      )
      if (maxOut > 0) {
        s.dischargeToLoadKwh = maxOut
        // Battery draw from stored energy to deliver maxOut to load:
        const batteryDraw = maxOut / params.roundTripEff
        socKwh = Math.max(0, socKwh - batteryDraw)
        s.gridImportKwh = Math.max(0, s.gridImportKwh - maxOut)
        s.slotCostEur = s.gridImportKwh * s.priceCtKwh / 100
      }
    }

    s.socKwhEnd = socKwh
  }

  // ---- Pass 3: sanity walk — ensure SoC continuity and no negative grid import ----
  for (let i = 0; i < N; i++) {
    const s = slots[i]
    if (s.gridImportKwh < 0) s.gridImportKwh = 0       // enforcement (safety)
    s.gridExportKwh = 0                                  // enforce export prohibition
    if (s.socKwhEnd < 0) s.socKwhEnd = 0
    if (s.socKwhEnd > params.usableKwh) s.socKwhEnd = params.usableKwh
    if (i > 0) slots[i].socKwhStart = slots[i-1].socKwhEnd
  }

  // ---- Roll up summary ----
  const baselineCostEur = slots.reduce((a, s) => a + s.baselineCostEur, 0)
  const optimizedCostEur = slots.reduce((a, s) => a + s.slotCostEur, 0)
  const avgPriceCtKwh = slots.reduce((a, s) => a + s.priceCtKwh, 0) / N
  const standbyCostEur = (params.standbyWatts * 24 / 1000) * avgPriceCtKwh / 100
  const savingsEur = baselineCostEur - optimizedCostEur - standbyCostEur

  // Split savings into arbitrage vs PV self-consumption:
  // arbitrage = value of discharge-to-load events charged from grid
  // pvSelfConsumption = value of (direct PV + PV→battery→load) vs no-PV baseline
  const arbitrageSavingsEur = slots.reduce((a, s) => {
    const fromGrid = s.chargeFromGridKwh
    return a + (s.dischargeToLoadKwh > 0 && fromGrid > 0 ? 0 : 0)  // placeholder
  }, 0)
  // Simpler attribution: arbitrage = Σ (dischargeToLoadKwh × priceCtKwh − equivalent grid-charge cost)
  // For now: assign all discharge-at-expensive-times × price as arbitrage,
  // and PV self-consumption = Σ(pvSelfKwh × priceCtKwh) + Σ(chargeFromPvKwh × downstream discharge price).
  const arbitrageSavingsEur2 = slots.reduce((a, s) => a + s.dischargeToLoadKwh * s.priceCtKwh / 100, 0)
    - slots.reduce((a, s) => a + s.chargeFromGridKwh * s.priceCtKwh / 100, 0)
  const pvSelfConsumptionValueEur = slots.reduce((a, s) => a + s.pvSelfKwh * s.priceCtKwh / 100, 0)
    + slots.reduce((a, s) => a + s.chargeFromPvKwh * params.roundTripEff * avgPriceCtKwh / 100, 0)

  const gridImportKwh = slots.reduce((a, s) => a + s.gridImportKwh, 0)
  const cyclesEquivalent = slots.reduce((a, s) => a + s.chargeFromGridKwh + s.chargeFromPvKwh, 0) / params.usableKwh

  return {
    slots,
    summary: {
      baselineCostEur,
      optimizedCostEur: optimizedCostEur + standbyCostEur,
      savingsEur,
      arbitrageSavingsEur: Math.max(0, arbitrageSavingsEur2),
      pvSelfConsumptionValueEur,
      standbyCostEur,
      gridImportKwh,
      gridExportKwh: 0,
      batteryCyclesEquivalent: cyclesEquivalent,
    }
  }
}

export function runBatteryYear(
  pricesByDate: Map<string, HourlyPrice[]>,
  pvProfile: number[],
  loadProfile: number[],
  pvKwhPerYear: number,
  annualLoadKwh: number,
  params: BatteryParams,
): AnnualBatteryResult {
  const monthlyAccum = new Map<string, {
    savings: number; arb: number; pv: number; standby: number; cycles: number;
  }>()
  let annualSavings = 0, annualArb = 0, annualPv = 0, annualStandby = 0
  let annualGridImport = 0, annualCycles = 0
  let socCarry = 0

  const sortedDates = Array.from(pricesByDate.keys()).sort()
  for (const date of sortedDates) {
    const prices = pricesByDate.get(date)
    if (!prices || prices.length === 0) continue

    // Find which hour-of-year range this date occupies
    const [yy, mm, dd] = date.split('-').map(Number)
    const dayStart = Date.UTC(yy, mm-1, dd)
    const yearStart = Date.UTC(yy, 0, 1)
    const hourOfYear = Math.floor((dayStart - yearStart) / 3_600_000)
    const slotHours = prices.length === 96 ? 0.25 : 1

    // Scale normalized profiles (sum=1.0/year) to actual per-slot kWh
    const pvPerSlot = new Array(prices.length)
    const loadPerSlot = new Array(prices.length)
    for (let i = 0; i < prices.length; i++) {
      const hourIdx = (hourOfYear + Math.floor(i * slotHours)) % 8760
      const fracPv = pvProfile[hourIdx] ?? 0
      const fracLoad = loadProfile[hourIdx] ?? 0
      // Split hourly profile value across QH slots
      pvPerSlot[i] = fracPv * pvKwhPerYear * slotHours
      loadPerSlot[i] = fracLoad * annualLoadKwh * slotHours
    }

    const { summary } = runBatteryDay(prices, pvPerSlot, loadPerSlot, params, socCarry)
    socCarry = 0  // reset each day — battery starts empty (conservative)

    const monthKey = `${yy}-${String(mm).padStart(2, '0')}`
    const bucket = monthlyAccum.get(monthKey) ?? { savings: 0, arb: 0, pv: 0, standby: 0, cycles: 0 }
    bucket.savings += summary.savingsEur
    bucket.arb += summary.arbitrageSavingsEur
    bucket.pv += summary.pvSelfConsumptionValueEur
    bucket.standby += summary.standbyCostEur
    bucket.cycles += summary.batteryCyclesEquivalent
    monthlyAccum.set(monthKey, bucket)

    annualSavings += summary.savingsEur
    annualArb += summary.arbitrageSavingsEur
    annualPv += summary.pvSelfConsumptionValueEur
    annualStandby += summary.standbyCostEur
    annualGridImport += summary.gridImportKwh
    annualCycles += summary.batteryCyclesEquivalent
  }

  const months: MonthlyBatteryResult[] = Array.from(monthlyAccum.entries())
    .sort((a, b) => a[0].localeCompare(b[0]))
    .map(([month, b]) => ({
      month,
      savingsEur: b.savings,
      arbitrageSavingsEur: b.arb,
      pvSelfConsumptionValueEur: b.pv,
      standbyCostEur: b.standby,
      cyclesEquivalent: b.cycles,
    }))

  return {
    annualSavingsEur: annualSavings,
    arbitrageSavingsEur: annualArb,
    pvSelfConsumptionValueEur: annualPv,
    standbyCostEur: annualStandby,
    gridImportKwh: annualGridImport,
    cyclesEquivalent: annualCycles,
    months,
  }
}

Key implementation notes:
- Use the 40th/60th percentile heuristic for cheap/expensive thresholds. The test for flat prices (all slots at 20ct) must pass: when cheapCutoff === expensiveCutoff, the "charge ≤ cheap" and "discharge ≥ expensive" conditions should not double-fire (ensure cheap and expensive are disjoint by strict < and > when cutoffs collide — i.e. use `s.priceCtKwh < cheapCutoff` if equal, OR accept that cycling is disabled when there is no spread).
