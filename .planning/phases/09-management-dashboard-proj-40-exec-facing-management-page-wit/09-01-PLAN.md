    export interface ManagementScenario {
      batteryCapacityKwh: number;      // default 60
      chargePowerKw: number;            // default 7
      plugInTime: string;               // "HH:MM", default "18:00"
      departureTime: string;            // "HH:MM", default "06:00"
      sessionsPerWeek: number;          // default 4
    }

    export interface MonthlyAggregate {
      year: number;                     // e.g. 2025
      month: number;                    // 1..12
      monthKey: string;                 // "YYYY-MM"
      avgSpreadCtKwh: number;           // mean (max - min in charging window) in ct/kWh
      energyPerSessionKwh: number;      // derived: chargePowerKw * windowHours, capped by battery
      sessionsInMonth: number;          // sessionsPerWeek * (days / 7)
      savingsEur: number;               // avgSpreadCtKwh/100 * energyPerSessionKwh * sessionsInMonth
      avgDayAheadCtKwh: number;         // mean of all QH prices in month, ct/kWh
    }

    export interface YoyDatum {
      monthKey: string;                 // "YYYY-MM"
      yearA: number;                    // older year totals, e.g. 2025
      yearB: number;                    // newer year totals, e.g. 2026
      valueA: number;                   // savings in EUR
      valueB: number;                   // savings in EUR
      deltaPct: number | null;          // (B-A)/A * 100, null if A == 0
    }

    export interface ExplainerData {
      monthKey: string;                 // month this explainer describes
      avgQhProfile: { qhIndex: number; ctKwh: number }[]; // length 96
      chargingWindow: { startQh: number; endQh: number };  // inclusive indices in 0..95
      baselineWindow: { startQh: number; endQh: number };  // dumb charging right after plug-in
      spreadCtKwh: number;
      energyPerSessionKwh: number;
      sessionsInMonth: number;
      reconciledSavingsEur: number;     // spreadCtKwh/100 * energyPerSessionKwh * sessionsInMonth
    }

    export interface ManagementDataset {
      schemaVersion: 1;
      generatedAt: string;              // ISO timestamp
      scenario: ManagementScenario;
      monthly: MonthlyAggregate[];
      explainer: ExplainerData;         // for the latest complete month
    }

    export const DEFAULT_MANAGEMENT_SCENARIO: ManagementScenario = {
      batteryCapacityKwh: 60,
      chargePowerKw: 7,
      plugInTime: "18:00",
      departureTime: "06:00",
      sessionsPerWeek: 4,
    };

    export const MANAGEMENT_STORAGE_KEY = "flexmon-management-scenario-v1";

    export const MANAGEMENT_DATA_URL = "/data/management-monthly.json";

    Required exports:

    ```typescript
    import type {
      ManagementScenario,
      MonthlyAggregate,
      YoyDatum,
      ExplainerData,
    } from "./management-config";

    export interface QhPricePoint { ts: string; price: number } // price in EUR/MWh

    // Convert "HH:MM" to quarter-hour index (0..95). "18:00" -> 72, "06:00" -> 24.
    export function hhmmToQhIndex(hhmm: string): number;

    // Build [startQh, endQh] for the charging window; wraps past midnight when departure < plug-in.
    // Returns indices inclusive; caller handles wrap (endQh < startQh => wraps).
    export function chargingWindowQh(scenario: ManagementScenario): { startQh: number; endQh: number };

    // Baseline "dumb" window: starts at plug-in, fills at charge power until battery target hit.
    export function baselineWindowQh(scenario: ManagementScenario): { startQh: number; endQh: number };

    // Energy per session: min(batteryCapacityKwh, chargePowerKw * windowHours).
    // Windows wrap; compute hour count correctly.
    export function energyPerSession(scenario: ManagementScenario): number;

    // Group prices by YYYY-MM and compute monthly aggregates.
    // - avgSpreadCtKwh: average over days in month of (max - min of QH prices inside chargingWindowQh), in ct/kWh.
    // - savingsEur: avgSpreadCtKwh/100 * energyPerSessionKwh * sessionsInMonth.
    // - sessionsInMonth: scenario.sessionsPerWeek * (daysInMonth / 7), rounded to 1 decimal stored as number.
    // - avgDayAheadCtKwh: mean across all QH points in the month, in ct/kWh.
    export function aggregateMonthly(
      qhPrices: QhPricePoint[],
      scenario: ManagementScenario
    ): MonthlyAggregate[];

    // Build YoY pairs for two years; align by month (1..12).
    export function computeYoy(
      monthly: MonthlyAggregate[],
      yearA: number,
      yearB: number
    ): YoyDatum[];

    // Build explainer for the given monthKey "YYYY-MM":
    // - avgQhProfile: for each qhIndex 0..95, average ct/kWh across all days in month.
    // - windows from scenario.
    // - spreadCtKwh = max(profile in charging window) - min(profile in charging window).
    // - reconciledSavingsEur = spreadCtKwh/100 * energyPerSessionKwh * sessionsInMonth.
    export function computeExplainer(
      qhPrices: QhPricePoint[],
      monthKey: string,
      scenario: ManagementScenario
    ): ExplainerData;

    // Reconciliation: returns the same product used above. Used by tests and UI to prove within 1%.
    export function reconcile(
      spreadCtKwh: number,
      energyPerSessionKwh: number,
      sessionsInMonth: number
    ): number;
    ```

    Guards:
    - Throw `Error("Invalid HH:MM: ...")` for malformed inputs.
    - Returns `[]` (or a zeroed explainer) when prices array is empty — never NaN.
    - Round savings to 2 decimals in output (keep internal precision).

    Use UTC throughout; build Date via `new Date(ts + "T12:00:00Z")` only when parsing YYYY-MM-DD strings (per codebase convention); SMARD QH timestamps are already ISO.
